        public void MoveAxis(string axis, int direction)
        {
            double delta = _state.Distance * direction;
            switch (axis)
            {
                case "X": _state.X = Math.Clamp(_state.X + delta, JogState.MinX, JogState.MaxX); break;
                case "Y": _state.Y = Math.Clamp(_state.Y + delta, JogState.MinY, JogState.MaxY); break;
                case "Z": _state.Z = Math.Clamp(_state.Z + delta, JogState.MinZ, JogState.MaxZ); break;
            }
            UpdatePositionDisplay();
        }

        private void UpdatePositionDisplay()
        {
            var values = new[] { _state.X, _state.Y, _state.Z };
            var limits = new[] {
                (JogState.MinX, JogState.MaxX),
                (JogState.MinY, JogState.MaxY),
                (JogState.MinZ, JogState.MaxZ)
            };

            for (int i = 0; i < 3; i++)
            {
                positionLabels[i].Text = values[i].ToString("F1");
                ApplyLimitWarning(positionLabels[i], values[i], limits[i].Item1, limits[i].Item2);
            }
        }

        private void ApplyLimitWarning(Label lbl, double value, double min, double max)
        {
            double range = max - min;
            double ratio = Math.Min(value - min, max - value) / range;

            if (ratio < 0.05) // 5% Ïù¥ÎÇ¥ - ÏúÑÌóò
            {
                lbl.BackColor = Color.FromArgb(61, 0, 0);
                lbl.ForeColor = Color.FromArgb(255, 82, 82);
            }
            else if (ratio < 0.10) // 10% Ïù¥ÎÇ¥ - Í≤ΩÍ≥†
            {
                lbl.BackColor = Color.FromArgb(61, 61, 0);
                lbl.ForeColor = Color.FromArgb(255, 235, 59);
            }
            else // Ï†ïÏÉÅ
            {
                lbl.BackColor = Color.FromArgb(10, 42, 10);
                lbl.ForeColor = Color.White;
            }
        }

    /// <summary>
    /// Í∞†Ìä∏Î¶¨ ÏÉÅÎåÄ ÏúÑÏπò Ï°∞ÍπÖ ÏÉÅÌÉú Í¥ÄÎ¶¨
    /// </summary>
    public class JogState
    {
        // === Ï∂ïÎ≥Ñ Î¶¨Î∞ã Í∞í (ÏÉà Í∞í Î∞òÏòÅ) ===
        public const double MaxX = 11905;
        public const double MinX = -1705;
        public const double MaxY = 5496.5;
        public const double MinY = -505.5;
        public const double MaxZ = -44;
        public const double MinZ = -1296;

        // === ÌòÑÏû¨ ÏúÑÏπò ===
        public double X { get; set; } = 10400;
        public double Y { get; set; } = 504.5;
        public double Z { get; set; } = 2795;

        // === Ïù¥Îèô ÏÑ§Ï†ï ===
        public int Distance { get; set; } = 100;  // mm
        public int Speed { get; set; } = 50;      // %

        /// <summary>
        /// Ï∂ïÎ≥Ñ Î¶¨Î∞ã Í∑ºÏ†ë ÎπÑÏú® Í≥ÑÏÇ∞ (0.0 ~ 1.0)
        /// </summary>
        public double GetLimitRatio(string axis)
        {
            double value, min, max;
            switch (axis)
            {
                case "X": value = X; min = MinX; max = MaxX; break;
                case "Y": value = Y; min = MinY; max = MaxY; break;
                case "Z": value = Z; min = MinZ; max = MaxZ; break;
                default: return 1.0;
            }
            double range = Math.Abs(max - min);
            double fromMin = Math.Abs(value - min);
            double fromMax = Math.Abs(max - value);
            return Math.Min(fromMin, fromMax) / range;
        }

        public WarningLevel GetWarningLevel(string axis)
        {
            double ratio = GetLimitRatio(axis);
            if (ratio < 0.05) return WarningLevel.Danger;
            if (ratio < 0.10) return WarningLevel.Warning;
            return WarningLevel.Normal;
        }
    }

    public enum WarningLevel
    {
        Normal,   // Ï†ïÏÉÅ
        Warning,  // Í≤ΩÍ≥† (10% Ïù¥ÎÇ¥)
        Danger    // ÏúÑÌóò (5% Ïù¥ÎÇ¥)
    }

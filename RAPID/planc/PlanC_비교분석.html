PROC rWarm_Up()
    ! 갠트리 X,Y,Z,R 축 순차 워밍업
    ! 속도를 점진적으로 증가 (10 → 25 deg/s, 300 → 250 mm/s)
    ! 설정된 시간(nWarmUpCycle) 동안 반복 실행

    vWarmUp_Speed:=[300,300,10,10];
    WHILE bWarmUpComplete=FALSE DO
        ! X축: ±150mm 왕복
        ! Y축: 0~150mm 왕복
        ! Z축: 0~100mm 왕복
        ! R축: ±90° 왕복
        MoveJgJ jgWarm_Up, vWarmUp_Speed, fine;
        vWarmUp_Speed.v_reax:=vWarmUp_Speed.v_reax+0.1;  ! 점진 가속
    ENDWHILE
ENDPROC
